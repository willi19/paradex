def get_camera_config() -> dict:
    cam_config_path = os.path.join(config_dir, "camera.json")
    if os.path.exists(cam_config_path):
        cam_info = json.load(open(cam_config_path, "r"))
    else:
        cam_info = {}
    return cam_info
